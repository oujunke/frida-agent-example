def print_object_methods(obj):  
    for attr_name in dir(obj):  
        attr_value = getattr(obj, attr_name)  
        if callable(attr_value):  
            print(f"{attr_name}()")

def get_and_encode_bytes(ea, size):  
    """  
    获取指定地址的字节并编码为 Base64  
    ea: 内存地址  
    size: 要获取的字节数  
    """  
    bytes_data = idaapi.get_bytes(ea, size)  
    if bytes_data:
        encoded_data = base64.b64encode(bytes_data).decode('utf-8')  
        print(f"Base64 encoded data at {hex(ea)}-endBy{hex(bytes_data[size-1])}: {encoded_data}")  
    else:  
        print(f"Failed to get bytes at {hex(ea)}")
